    <p class="calc-subtitle">근무일수를 기준으로 급여를 일할 계산합니다 (30일 기준)</p>

    <div class="field">
      <label>계산 대상 월</label>
      <input type="month" id="targetMonth" />
    </div>

    <div class="field">
      <label>월 기본급</label>
      <input type="text" id="baseSalary" placeholder="예: 3,000,000" oninput="fmtInput(this)" />
    </div>

    <div class="field">
      <label>수당</label>
      <small>식대 등 수당을 + 버튼으로 추가하세요.</small>
      <div id="itemList"></div>
      <button type="button" class="add-btn" onclick="addItem()">+ 항목 추가</button>
    </div>

    <div class="field">
      <label>근무일 체크</label>
      <small>실제 근무한 날만 선택하세요. (기본값: 전체 선택)</small>
      <div class="cal-head">
        <span class="count" id="dayCount">근무일: 0일</span>
        <button type="button" class="toggle-all" id="toggleAll" onclick="toggleAll()">전체 해제</button>
      </div>
      <div class="calendar"><div class="cal-grid" id="calGrid"></div></div>
    </div>

    <button type="button" class="calc-submit" onclick="calculate()">계산하기</button>

    <div id="result">
      <p class="result-label">계산 결과</p>
      <div class="result-list" id="resultList"></div>

<script>
  // ===== 유틸 =====
  function fmt(n){ return Math.round(n).toLocaleString('ko-KR') + '원'; }
  function fmtInput(el){ const v=el.value.replace(/[^0-9]/g,''); el.value = v ? Number(v).toLocaleString('ko-KR') : ''; }
  function toNumber(str){ return Number((str||'').replace(/[^0-9]/g,'')) || 0; }

  const BASE_DAYS = 30;          // ← 일할 기준 일수 (30일 고정)
  let selectedDays = new Set();
  let daysInMonth = 0;
  let curMonth = 0;

  // ===== 수당 항목 =====
  function addItem(name='', amount=''){
    const row=document.createElement('div'); row.className='item-row';
    row.innerHTML='<input type="text" class="name" placeholder="항목명 (예: 식대)" />'+
      '<input type="text" class="amount" placeholder="금액" oninput="fmtInput(this)" />'+
      '<button type="button" class="del" onclick="this.parentElement.remove()">×</button>';
    row.querySelector('.name').value=name; row.querySelector('.amount').value=amount;
    document.getElementById('itemList').appendChild(row);
  }

  // ===== 달력 =====
  function buildCalendar(){
    const [year, month] = document.getElementById('targetMonth').value.split('-').map(Number);
    curMonth = month;
    daysInMonth = new Date(year, month, 0).getDate();
    const firstDow = new Date(year, month-1, 1).getDay();

    selectedDays = new Set();
    for(let d=1; d<=daysInMonth; d++) selectedDays.add(d);   // 기본: 전체 선택

    const grid=document.getElementById('calGrid'); grid.innerHTML='';
    ['일','월','화','수','목','금','토'].forEach((dw,i)=>{
      const el=document.createElement('div'); el.className='cal-dow'+(i===0?' sun':i===6?' sat':''); el.textContent=dw; grid.appendChild(el);
    });
    for(let i=0;i<firstDow;i++){ const b=document.createElement('div'); b.className='cal-day empty'; grid.appendChild(b); }

    for(let d=1; d<=daysInMonth; d++){
      const dow=new Date(year, month-1, d).getDay();
      const btn=document.createElement('div');
      btn.className='cal-day on'+(dow===0?' sun':dow===6?' sat':'');
      btn.textContent=d;
      btn.onclick=()=>{
        if(selectedDays.has(d)){ selectedDays.delete(d); btn.classList.remove('on'); }
        else { selectedDays.add(d); btn.classList.add('on'); }
        updateCount();
      };
      grid.appendChild(btn);
    }
    updateCount();
    document.getElementById('toggleAll').textContent='전체 해제';
  }

  function updateCount(){ document.getElementById('dayCount').textContent='근무일: '+selectedDays.size+'일'; }

  function toggleAll(){
    const btn=document.getElementById('toggleAll');
    const allDays=document.querySelectorAll('.cal-day:not(.empty)');
    if(selectedDays.size>0){ selectedDays.clear(); allDays.forEach(el=>el.classList.remove('on')); btn.textContent='전체 선택'; }
    else { allDays.forEach((el,i)=>{ selectedDays.add(i+1); el.classList.add('on'); }); btn.textContent='전체 해제'; }
    updateCount();
  }

  // ===== 계산 =====
  function calculate(){
    const workedDays = selectedDays.size;
    const ratio = workedDays / BASE_DAYS;          // 30일 기준
    const list=document.getElementById('resultList'); list.innerHTML='';
    const add=(label,val,cls='')=>list.insertAdjacentHTML('beforeend',`<div class="result-row ${cls}"><span>${label}</span><span>${val}</span></div>`);

    add(`${curMonth}월 근무일 (기준 ${BASE_DAYS}일)`, workedDays+'일');

    const base=toNumber(document.getElementById('baseSalary').value);
    let total=Math.round(base*ratio);
    add('기본급', fmt(total));

    document.querySelectorAll('#itemList .item-row').forEach(row=>{
      const name=row.querySelector('.name').value.trim()||'수당';
      const amt=toNumber(row.querySelector('.amount').value);
      if(amt===0) return;
      const p=Math.round(amt*ratio); total+=p;
      add(name, fmt(p));
    });

    add(`${curMonth}월 급여`, fmt(total), 'total');
    document.getElementById('result').style.display='block';
  }

  // ===== 초기화 =====
  (function init(){
    const now=new Date();
    const ym=now.getFullYear()+'-'+String(now.getMonth()+1).padStart(2,'0');
    const mi=document.getElementById('targetMonth'); mi.value=ym;
    mi.addEventListener('change', buildCalendar);
    addItem('식대','');
    buildCalendar();
  })();
</script>
